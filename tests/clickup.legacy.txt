import { describe, expect, it, vi } from "vitest";
import { ClickUpService } from "../src/clickup/service.js";
import { createAnalysis, createConfig, createProblem } from "./helpers.js";

describe("ClickUpService", () => {
  it("detects duplicates by composite problem key", async () => {
    const client = {
      getTasks: vi.fn().mockResolvedValue({
        tasks: [{ id: "1", name: "task", description: "Problem Key: 1::two-sum" }],
        lastPage: true
      }),
      createTask: vi.fn()
    };

    const service = new ClickUpService(createConfig(), client as never);
    await expect(service.hasProblem("1::two-sum")).resolves.toBe(true);
  });

  it("maps question number and slug into task payload", async () => {
    const client = {
      getTasks: vi.fn(),
      createTask: vi.fn().mockResolvedValue({ id: "1" }),
      getAccessibleCustomFields: vi.fn().mockResolvedValue({
        fields: [
          { id: "field-question-number", name: "Question Number", type: "short_text" },
          { id: "field-question-slug", name: "Question Slug", type: "short_text" },
          {
            id: "field-difficulty",
            name: "Difficulty",
            type: "drop_down",
            type_config: { options: [{ id: "opt-medium", name: "Medium" }, { id: "opt-easy", name: "Easy" }] }
          },
          {
            id: "field-topic-tags",
            name: "Topic Tags",
            type: "labels",
            type_config: {
              options: [
                { id: "opt-array", label: "Array" },
                { id: "opt-hash-table", label: "Hash Table" }
              ]
            }
          },
          { id: "field-date-solved", name: "Date Solved", type: "date" }
        ]
      }).mockImplementation(async () => [
        { id: "field-question-number", name: "Question Number", type: "short_text" },
        { id: "field-question-slug", name: "Question Slug", type: "short_text" },
        {
          id: "field-difficulty",
          name: "Difficulty",
          type: "drop_down",
          type_config: { options: [{ id: "opt-medium", name: "Medium" }, { id: "opt-easy", name: "Easy" }] }
        },
        {
          id: "field-topic-tags",
          name: "Topic Tags",
          type: "labels",
          type_config: {
            options: [
              { id: "opt-array", label: "Array" },
              { id: "opt-hash-table", label: "Hash Table" }
              ]
            }
          },
        { id: "field-date-solved", name: "Date Solved", type: "date" }
      ]),
      setCustomFieldValue: vi.fn().mockResolvedValue(undefined)
    };

    const service = new ClickUpService(createConfig(), client as never);
    await service.createProblemTask(createProblem(), createAnalysis());

    const payload = client.createTask.mock.calls[0][0];
    expect(payload.name).toBe("#1 Two Sum");
    expect(payload.description).toContain("Question Number: 1");
    expect(payload.description).toContain("Question Slug: two-sum");
    expect(payload.description).toContain("## Problem URL");
    expect(payload.description).toContain("https://leetcode.com/problems/two-sum/");
    expect(payload.description).toContain("## My Solution");
    expect(payload.description).toContain("function twoSum() {}");
    expect(payload.description).toContain("## AI Summary");
    expect(client.setCustomFieldValue.mock.calls).toEqual(
      expect.arrayContaining([
        ["1", "field-question-number", { value: "1" }],
        ["1", "field-question-slug", { value: "two-sum" }],
        ["1", "field-difficulty", { value: "opt-easy" }],
        ["1", "field-topic-tags", { value: ["opt-array", "opt-hash-table"] }]
      ])
    );
    expect(client.setCustomFieldValue.mock.calls).toHaveLength(5);
  });

  it("stops duplicate scanning on an empty page", async () => {
    const client = {
      getTasks: vi.fn().mockResolvedValue({
        tasks: [],
        lastPage: false
      }),
      createTask: vi.fn()
    };

    const service = new ClickUpService(createConfig(), client as never);
    await expect(service.hasProblem("1::two-sum")).resolves.toBe(false);
    expect(client.getTasks).toHaveBeenCalledTimes(1);
  });
});
